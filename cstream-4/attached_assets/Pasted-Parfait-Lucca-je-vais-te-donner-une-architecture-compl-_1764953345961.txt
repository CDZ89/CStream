## ðŸŽ¨ Frontend (React)

### Composant `ShinyText` avec logo SVG

```tsx
import './ShinyText.css';
import { ReactComponent as WebIcon } from './web.svg';
import { ReactComponent as SearchIcon } from './search.svg';
import { ReactComponent as ImageIcon } from './image.svg';

interface ShinyTextProps {
  text: string;
  disabled?: boolean;
  speed?: number;
  className?: string;
  svgIcon?: React.ReactNode;
}

const ShinyText: React.FC<ShinyTextProps> = ({
  text,
  disabled = false,
  speed = 5,
  className = '',
  svgIcon
}) => {
  const animationDuration = `${speed}s`;

  return (
    <div className={`shiny-text-wrapper ${className}`} style={{ animationDuration }}>
      {svgIcon && <span className="shiny-icon">{svgIcon}</span>}
      <span className={`shiny-text ${disabled ? 'disabled' : ''}`}>
        {text}
      </span>
    </div>
  );
};

export default ShinyText;
```
