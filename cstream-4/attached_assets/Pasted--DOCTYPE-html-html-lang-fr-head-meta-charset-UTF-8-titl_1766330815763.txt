  <script>
    // ============================
    // PARTICULES MINIMALISTES
    // ============================
    const canvas = document.getElementById('spark-canvas');
    const ctx = canvas.getContext('2d');

    let particles = [];
    const PARTICLE_COUNT = 90;      // nombre d'étincelles
    const MAX_SPEED = 0.15;        // vitesse max (lent = pro)
    const MAX_SIZE = 1.4;          // taille max (petit = clean)

    function resizeCanvas() {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    }

    window.addEventListener('resize', () => {
      resizeCanvas();
      initParticles();
    });

    function random(min, max) {
      return Math.random() * (max - min) + min;
    }

    function initParticles() {
      particles = [];
      for (let i = 0; i < PARTICLE_COUNT; i++) {
        particles.push({
          x: random(0, canvas.width),
          y: random(0, canvas.height),
          vx: random(-MAX_SPEED, MAX_SPEED),
          vy: random(-MAX_SPEED, MAX_SPEED),
          r: random(0.4, MAX_SIZE),
          alpha: random(0.25, 0.8),
          alphaSpeed: random(0.0005, 0.0015)
        });
      }
    }

    function updateParticles() {
      for (let p of particles) {
        p.x += p.vx;
        p.y += p.vy;
        p.alpha += p.alphaSpeed;

        if (p.alpha > 0.9 || p.alpha < 0.2) {
          p.alphaSpeed *= -1;
        }

        // wrap tranquille sur les bords (pas de pop brusque)
        if (p.x < -20) p.x = canvas.width + 20;
        if (p.x > canvas.width + 20) p.x = -20;
        if (p.y < -20) p.y = canvas.height + 20;
        if (p.y > canvas.height + 20) p.y = -20;
      }
    }

    function drawParticles() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      for (let p of particles) {
        const gradient = ctx.createRadialGradient(
          p.x, p.y, 0,
          p.x, p.y, p.r * 4
        );
        gradient.addColorStop(0, `rgba(248, 250, 252, ${p.alpha})`);
        gradient.addColorStop(1, `rgba(248, 250, 252, 0)`);

        ctx.fillStyle = gradient;
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.r * 4, 0, Math.PI * 2);
        ctx.fill();
      }
    }

    function loop() {
      updateParticles();
      drawParticles();
      requestAnimationFrame(loop);
    }

    resizeCanvas();
    initParticles();
    loop();
  </script>
