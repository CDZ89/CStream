interface MediaPageProps {
  title: string;
  posterUrl: string;
  logoUrl: string;

export default function MediaPage({ title, posterUrl, logoUrl }: MediaPageProps) {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div className="min-h-screen bg-neutral-900 text-white flex flex-col items-center p-6">
      {/* Logo */}
      <img
        src={logoUrl}
        alt="TMDB Logo"
        className="w-32 h-auto mb-6"
      />

      {/* Titre */}
      <h1 className="text-3xl font-bold mb-6">{title}</h1>

      {/* Affiche */}
      <div className="cursor-pointer" onClick={() => setIsOpen(true)}>
        <img
          src={posterUrl}
          alt={title}
          className="w-64 md:w-80 rounded-lg shadow-lg hover:scale-105 transition-transform"
        />
      </div>

      {/* Modal Lightbox */}
      {isOpen && (
        <div
          className="fixed inset-0 bg-black bg-opacity-80 flex items-center justify-center z-50"
          onClick={() => setIsOpen(false)}
        >
          <img
            src={posterUrl}
            alt={title}
            className="max-w-[90%] max-h-[90%] rounded-lg shadow-2xl"
          />
        </div>
      )}
    </div>
  );
}
