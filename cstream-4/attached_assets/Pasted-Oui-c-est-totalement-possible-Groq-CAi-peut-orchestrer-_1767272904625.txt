import fetch from "node-fetch";

const TMDB_BASE = "https://api.themoviedb.org/3";
const TMDB_TOKEN = process.env.TMDB_API_TOKEN!; // Bearer v4

export async function tmdbSearch(req, res) {
  const { q, lang = "fr-FR", page = "1" } = req.query;
  const url = `${TMDB_BASE}/search/movie?query=${encodeURIComponent(q)}&language=${lang}&include_adult=false&page=${page}`;
  const r = await fetch(url, { headers: { Authorization: `Bearer ${TMDB_TOKEN}` } });
  res.status(r.status).json(await r.json());
}

export async function tmdbDetails(req, res) {
  const { id } = req.params;
  const { lang = "fr-FR" } = req.query;
  const append = "images,credits,recommendations,release_dates";
  const url = `${TMDB_BASE}/movie/${id}?language=${lang}&append_to_response=${append}`;
  const r = await fetch(url, { headers: { Authorization: `Bearer ${TMDB_TOKEN}` } });
  res.status(r.status).json(await r.json());
}
