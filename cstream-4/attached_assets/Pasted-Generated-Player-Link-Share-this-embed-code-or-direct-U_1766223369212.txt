Watch Progress
Track your users' watch progress across movies and TV shows. This feature enables Continue Watching functionality on your website.
Add Event Listener
Place this inside your React or Next.js useEffect where the player iframe exists.

Script
Copy
window.addEventListener('message', (event) => {
  if (event.origin !== 'https://player.vidify.top') return;

  if (event.data?.type === 'WATCH_PROGRESS') {
    const { mediaId, currentTime, duration } = event.data.data;
    localStorage.setItem(`progress_${mediaId}`, JSON.stringify({
      currentTime, duration, lastWatched: Date.now()
    }));
  }
});

// Complete progress tracking implementation
window.addEventListener('message', (event) => {
  if (event.origin !== 'https://player.vidify.top') return;

  if (event.data?.type === 'WATCH_PROGRESS') {
    const { mediaId, eventType, currentTime, duration } = event.data.data;

    // Handle the event
    console.log('Watch Progress:', { mediaId, eventType, currentTime, duration });

    // Store progress data
    const progressData = {
      currentTime,
      duration,
      lastWatched: Date.now(),
      eventType
    };

    localStorage.setItem(\`progress_${mediaId}\`, JSON.stringify(progressData));

    // Optional: Handle specific events
    if (eventType === 'play') {
      console.log('Video started playing');
    } else if (eventType === 'pause') {
      console.log('Video paused');
    }
  }
});
