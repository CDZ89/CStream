  <script>
    const sidebar = document.getElementById('sidebar');
    const toggleSidebarBtn = document.getElementById('toggleSidebar');
    const convos = document.getElementById('convos');
    const newConvoBtn = document.getElementById('newConvo');
    const welcome = document.getElementById('welcome');
    const chat = document.getElementById('chat');
    const messages = document.getElementById('messages');
    const input = document.getElementById('input');
    const send = document.getElementById('send');
    const startInput = document.getElementById('startInput');

    let conversations = [];
    let currentConvo = null;

    toggleSidebarBtn.onclick = () => sidebar.classList.toggle('hidden');

    function formatTime() {
      const now = new Date();
      return now.toLocaleDateString('fr-FR') + " ‚Ä¢ " + now.toLocaleTimeString('fr-FR',{hour:'2-digit',minute:'2-digit'});
    }

    function addMessage(text, role='user') {
      const bubble = document.createElement('div');
      bubble.className = `bubble ${role}`;
            bubble.innerHTML = `
        <strong>${role === 'user' ? 'Moi' : 'CAi'}</strong> 
        ‚Ä¢ <small>${formatTime()}</small><br>${text}
      `;
      messages.appendChild(bubble);
      messages.scrollTop = messages.scrollHeight;
      if (currentConvo) {
        currentConvo.msgs.push({ role, text, time: formatTime() });
      }
    }

    function renderConvos() {
      convos.innerHTML = '';
      conversations.forEach((c, i) => {
        const div = document.createElement('div');
        div.className = 'convo';
        div.innerHTML = `
          <span>${c.name}</span>
          <button onclick="renameConvo(${i})">‚úèÔ∏è</button>
          <button onclick="deleteConvo(${i})">üóëÔ∏è</button>
        `;
        div.onclick = () => loadConvo(i);
        convos.appendChild(div);
      });
    }

    function newConvo(name = null) {
      const convo = { name: name || 'Conversation ' + (conversations.length + 1), msgs: [] };
      conversations.push(convo);
      currentConvo = convo;
      messages.innerHTML = '';
      welcome.style.display = 'none';
      chat.style.display = 'flex';
      renderConvos();
    }

    function loadConvo(i) {
      currentConvo = conversations[i];
      messages.innerHTML = '';
      welcome.style.display = 'none';
      chat.style.display = 'flex';
      currentConvo.msgs.forEach(m => {
        const bubble = document.createElement('div');
        bubble.className = `bubble ${m.role}`;
        bubble.innerHTML = `
          <strong>${m.role === 'user' ? 'Moi' : 'CAi'}</strong> 
          ‚Ä¢ <small>${m.time}</small><br>${m.text}
        `;
        messages.appendChild(bubble);
      });
    }

    function renameConvo(i) {
      const newName = prompt("Nouveau nom :", conversations[i].name);
      if (newName) {
        conversations[i].name = newName;
        renderConvos();
      }
    }

    function deleteConvo(i) {
      conversations.splice(i, 1);
      currentConvo = null;
      messages.innerHTML = '';
      chat.style.display = 'none';
      welcome.style.display = 'flex';
      renderConvos();
    }

    // Bouton nouvelle conversation
    newConvoBtn.onclick = () => newConvo();

    // Input accueil
    startInput.addEventListener('keydown', e => {
      if (e.key === 'Enter') {
        const val = startInput.value.trim();
        if (val) {
          newConvo(val);
          startInput.value = '';
        }
      }
    });

    // Envoi de message
    send.addEventListener('click', () => {
      const text = input.value.trim();
      if (!text) return;
      addMessage(text, 'user');
      input.value = '';
      // Simuler r√©ponse IA
      setTimeout(() => {
        addMessage("R√©ponse premium en verre liquide üíé", 'ai');
      }, 1000);
    });
