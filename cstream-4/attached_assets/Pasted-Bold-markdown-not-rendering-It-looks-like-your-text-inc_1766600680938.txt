If you‚Äôre in React
tsx
import { useMemo } from 'react';
import DOMPurify from 'dompurify';
import { marked } from 'marked';

export function Markdown({ md }: { md: string }) {
  const html = useMemo(() => DOMPurify.sanitize(marked.parse(md, { gfm: true, breaks: true })), [md]);
  return <div className="prose prose-invert" dangerouslySetInnerHTML={{ __html: html }} />;
}

If you see literal ** in the UI, you‚Äôre rendering plain text; parse it.
