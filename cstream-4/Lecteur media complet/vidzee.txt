                updateEmbed('tv', idInput.value, seasonInput.value, episodeInput.value);

            function updateEmbed(type, id, season = null, episode = null) {
                const url = type === 'movie' 
                    ? `https://player.vidzee.wtf/embed/movie/${id}`
                    : `https://player.vidzee.wtf/embed/tv/${id}/${season}/${episode}`;
                embedInput.value = url;
                iframe.src = url;
            }

            [idInput, seasonInput, episodeInput].forEach(input => {
                input.addEventListener('change', () => {
                    if (seriesBtn.classList.contains('active')) {
                        updateEmbed('tv', idInput.value, seasonInput.value, episodeInput.value);
                    } else {
                        updateEmbed('movie', idInput.value);
                    }
                });
            });

            // Copy Buttons
            document.querySelectorAll('.copy-btn').forEach(button => {
                button.addEventListener('click', function() {
                    let text = '';
                    const code = this.closest('.copy-text').querySelector('code');
                    const input = this.closest('.copy-text').querySelector('input.text');
                    text = code ? code.textContent : (input ? input.value : '');

                    navigator.clipboard.writeText(text).then(() => {
                        toast.classList.add('show');
                        setTimeout(() => toast.classList.remove('show'), 2000);

                        document.body.removeChild(textarea);

                    });
                });
            });
