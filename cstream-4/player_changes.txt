+                  variant="ghost"
+                  size="icon"

+                  title="Next Episode"

+                </div>
+              </motion.div>

+        </AnimatePresence>
+
+        {/* Video Iframe */}
+        <iframe
+          ref={iframeRef}
+          key={iframeKey}
+          src={videoUrl}

+          onError={handleIframeError}
+          title={title || "Video Player"}
+        />
+      </div>
+    </div>
+  );
+};
